// ./Domain/Tokens/ITokenAnimator.cs

    public interface ITokenAnimator
    {
        Task Animate(TokenResolution resolution, TokenVisualContext context, int sourceClientIndex);

    public abstract class TokenAnimator<TResolution> : ITokenAnimator where TResolution : TokenResolution

        protected abstract Task Animate(TResolution resolution, TokenVisualContext context, int sourceClientIndex);
    }

    {
        public static GameConnection Singleton { get; private set; }

        [Rpc(SendTo.Server)]

        [Rpc(SendTo.SpecifiedInParams)]

        [Rpc(SendTo.Server)]
        public void MessageServerMatchClientReadyRpc(ulong matchId, RpcParams rpc = default);
        [Rpc(SendTo.SpecifiedInParams)]
        public void BroadcastClientMatchStartedRpc(GameState gameState, RpcParams rpc = default);
        // -------------------------------------- Match Flow
        [Rpc(SendTo.Server)]
        public void MessageServerMatchPlayTokenRpc(TokenIntentMessage intentMsg, RpcParams rpc = default);
        [Rpc(SendTo.SpecifiedInParams)]
        public void BroadcastClientsMatchPlayTokenResolvedRpc(TokenResolutionMessage resolution, RpcParams rpc = default);
        [Rpc(SendTo.SpecifiedInParams)]
        public void BroadcastClientsMatchStartTurnRpc(int activeClientIndex, RpcParams rpc = default);
        [Rpc(SendTo.SpecifiedInParams)]
        public void BroadcastClientsMatchEndRoundRpc(int p0Score, int p1Score, RpcParams rpc = default);
    }
